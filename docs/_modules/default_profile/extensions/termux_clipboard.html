[docs]    @line_magic
    def termux_clipboard_get(self):
        if not shutil.which("termux-clipboard-get"):
            return
        p = subprocess.run(["termux-clipboard-get"], stdout=subprocess.PIPE)
        text = p.stdout
        return text
